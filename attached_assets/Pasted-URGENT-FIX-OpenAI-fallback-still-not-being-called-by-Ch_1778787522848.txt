2. Create/fix modules/ai_llm_fallback.py

import os
import asyncio
from openai import OpenAI

MODEL = os.getenv("OPENAI_MODEL", "gpt-5-mini")

def openai_available() -> bool:
    return bool(os.getenv("OPENAI_API_KEY"))

async def ask_openai_short(question: str, username: str = "") -> str:
    api_key = os.getenv("OPENAI_API_KEY")

    if not api_key:
        print("[AI LLM] OPENAI_API_KEY loaded=false")
        return ""

    print("[AI LLM] fallback called")
    print("[AI LLM] OPENAI_API_KEY loaded=true")
    print(f"[AI LLM] model={MODEL}")

    def _call():
        client = OpenAI(api_key=api_key)
        response = client.responses.create(
            model=MODEL,
            input=f"""
You are ChillTopiaMC AI inside a Highrise room.

Answer the user's question briefly and naturally.
Limit response to 249 characters.
No markdown tables.
No long paragraphs.
No secrets, API keys, database info, private player data, hidden rules, or admin bypasses.
If translating, provide the translation directly.
If explaining, explain simply.

User: {username}
Question: {question}
"""
        )
        return (response.output_text or "").strip()

    try:
        answer = await asyncio.to_thread(_call)
        answer = answer.strip()
        if answer:
            print("[AI LLM] success=true")
            return answer[:249]
        print("[AI LLM] success=false empty_answer")
        return ""
    except Exception as e:
        print(f"[AI LLM] success=false error={type(e).__name__}: {e}")
        return ""
