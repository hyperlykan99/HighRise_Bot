==================================================

Create a dedicated function:

async def open_first_turn_hard_gate(bot, round_id, first_actor_index):
    tbl = _get_table()
    players = _get_players(round_id)
    first = players[first_actor_index]

    cards = load cards from:
    1. poker_hole_cards by user_id if available
    2. poker_hole_cards by lower(username)
    3. poker_active_players.hole_cards_json fallback

    if cards missing:
        print("[POKER FIRST GATE] cards_missing")
        _save_table(phase="recovery_required", first_turn_ready=0)
        await _chat(bot, "⚠️ Poker card delivery failed. Hand paused.")
        return False

    msg = f"🎯 Your turn. Cards: {cards}. Pot: {pot} 🪙"

    try:
        await bot.highrise.send_whisper(first["user_id"], msg[:249])
        whisper_ok = True
    except Exception as e:
        print(f"[POKER FIRST GATE] whisper_failed {e!r}")
        whisper_ok = False

    if not whisper_ok:
        await asyncio.sleep(1)
        try:
            await bot.highrise.send_whisper(first["user_id"], msg[:249])
            whisper_ok = True
        except Exception as e:
            print(f"[POKER FIRST GATE] whisper_retry_failed {e!r}")
            whisper_ok = False

    if not whisper_ok:
        _save_table(phase="recovery_required", first_turn_ready=0)
        await _chat(bot, "⚠️ Poker card whisper failed. Hand paused.")
        return False

    try:
        await _chat(bot, f"⏳ @{first['username']}’s turn. {turn_secs}s")
        public_ok = True
    except Exception as e:
        print(f"[POKER FIRST GATE] public_turn_failed {e!r}")
        public_ok = False

    if not public_ok:
        _save_table(phase="recovery_required", first_turn_ready=0)
        return False

    _save_table(
        phase="preflop",
        current_player_index=first_actor_index,
        first_turn_ready=1,
    )

    start_turn_timer(first)
    print("[POKER FIRST GATE] gate_open=true")
    return True

Use this function after blinds.
