Required channel logic:

def choose_reply_channel(response_type, knowledge_level="public", contains_private=False):
    mode = get_reply_mode()

    if contains_private:
        return "whisper"

    if knowledge_level in ("PLAYER_PRIVATE", "STAFF", "ADMIN", "OWNER", "private", "staff", "admin", "owner"):
        return "whisper"

    if response_type in ("confirmation", "billing_error", "permission_denied", "owner_debug", "private_data"):
        return "whisper"

    if mode == "public":
        return "public"

    if mode == "whisper":
        return "whisper"

    # smart mode
    if response_type in ("general", "public_info", "guide", "openai_answer", "cost_info"):
        return "public"

    return "whisper"

Required _send behavior:
async def _send(bot, user, message, response_type="general", knowledge_level="public", contains_private=False):
    channel = choose_reply_channel(response_type, knowledge_level, contains_private)

    print(f"[AI REPLY] mode={get_reply_mode()} channel={channel} response_type={response_type} private={contains_private}")

    if channel == "public":
        try:
            await bot.highrise.chat(message[:249])
            return
        except Exception as e:
            print(f"[AI REPLY] public failed: {e}")

    try:
        await bot.highrise.send_whisper(user.id, message[:249])
    except Exception as e:
        print(f"[AI REPLY] whisper failed: {e}")
