Step 2 — Fix AI trigger stripping

Find or create strip_ai_trigger(message).

Use this exact logic or equivalent:

def strip_ai_trigger(message: str) -> str:
    if not message:
        return ""

    original = message.strip()
    lowered = original.lower()

    prefixes = [
        "ai",
        "@chilltopiamc",
        "chilltopiamc",
        "chilltopia",
        "assistant",
        "bot",
    ]

    for prefix in prefixes:
        if lowered == prefix:
            return ""

        if lowered.startswith(prefix + " "):
            return original[len(prefix):].strip()

        if lowered.startswith(prefix + ","):
            return original[len(prefix):].lstrip(" ,:?!").strip()

        if lowered.startswith(prefix + ":"):
            return original[len(prefix):].lstrip(" ,:?!").strip()

        if lowered.startswith(prefix + "?"):
            return original[len(prefix):].lstrip(" ,:?!").strip()

        if lowered.startswith(prefix + "!"):
            return original[len(prefix):].lstrip(" ,:?!").strip()

    return original

def get_current_bot_name(self=None) -> str:
    candidates = []

    try:
        if self and hasattr(self, "username"):
            candidates.append(self.username)
    except:
        pass

    try:
        if self and hasattr(self, "bot_name"):
            candidates.append(self.bot_name)
    except:
        pass

    try:
        if self and hasattr(self, "config") and hasattr(self.config, "BOT_NAME"):
            candidates.append(self.config.BOT_NAME)
    except:
        pass

    try:
        import os
        candidates.append(os.getenv("BOT_NAME"))
        candidates.append(os.getenv("HIGHRISE_BOT_NAME"))
    except:
        pass

    for name in candidates:
        if name and str(name).strip():
            return str(name).strip()

    return "ChillTopiaMC"
